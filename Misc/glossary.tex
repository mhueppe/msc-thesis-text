
\newglossaryentry{overfit}
{
	name=overfitting,
	description= Overfitting occurs when a model performs extremely well on the training data but fails to generalize to unseen data. This phenomenon is often attributed to the model's complexity. If a model is too complex in terms of its number of parameters it adapts neatly to the training data instead of learning the underlying features \citep{Hawkins2004}
}
